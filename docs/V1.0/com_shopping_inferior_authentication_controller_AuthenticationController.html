用户注册

请求URL

/user/addUser POST

请求体 application/json

{
	"userAccount":"string //用户账号",
	"pwd":"string //用户密码",
	"userName":"string //用户姓名",
	"userStatus":"int //用户状态，只有卖家需要，1为通过审核，0为未审核",
	"userPhone":"string //用户电话号码",
	"userRole":"string //用户角色，为seller/buyer/admin",
	"address":"string //用户默认地址"
}

返回结果

{
	"code":"string //响应代码，400为错误，200为正确",
	"msg":"string //响应提示信息,若为搜索请求,则为页码总数",
	"data":{}
}

用户登录

请求URL

/user/checkLogin GET

请求参数 application/x-www-form-urlencoded

参数名 类型 必须 描述
account string 是 用户账号
pwd string 是 用户密码

返回结果

{
	"code":"string //响应代码，400为错误，200为正确",
	"msg":"string //响应提示信息,若为搜索请求,则为页码总数",
	"data":{
		"id":"int //主键，用户编号",
		"userName":"string //用户姓名",
		"userStatus":"int //用户状态，1为已审核，0为未审核",
		"userPhone":"string //用户电话号码",
		"roleList":[{
			"id":"int //主键，关联类编号",
			"roleId":"int //角色编号",
			"accountId":"int //账号编号"
		}],
		"address":"string //用户默认地址",
		"token":"string //用户当前token"
	}
}
