#import <Foundation/Foundation>;

@interface Result : NSObject

//响应代码，400为错误，200为正确
@property (nonatomic, assign) NSString code;
//响应提示信息,若为搜索请求,则为页码总数
@property (nonatomic, assign) NSString msg;
//响应数据
@property (nonatomic, strong) Address[] data;


@end



#import <Foundation/Foundation>;

@interface Address : NSObject

//主键，地址编号
@property (nonatomic, strong) NSInteger id;
//用户编号
@property (nonatomic, strong) NSInteger userId;
//具体地址内容
@property (nonatomic, assign) NSString content;
//地址状态，1为默认地址，0为备用地址
@property (nonatomic, strong) NSInteger addressStatus;


@end
