public class Result implements Serializable{

    private String code; //响应代码，400为错误，200为正确
    private String msg; //响应提示信息,若为搜索请求,则为页码总数
    private Classes[] data; //响应数据

    public String getCode(){
        return code;
    }

    public void setCode(String code){
        this.code = code;
    }

    public String getMsg(){
        return msg;
    }

    public void setMsg(String msg){
        this.msg = msg;
    }

    public Classes[] getData(){
        return data;
    }

    public void setData(Classes[] data){
        this.data = data;
    }

}

public class Classes implements Serializable{

    private String id; //主键，类别编号
    private String className; //类别名

    public String getId(){
        return id;
    }

    public void setId(String id){
        this.id = id;
    }

    public String getClassName(){
        return className;
    }

    public void setClassName(String className){
        this.className = className;
    }

}
